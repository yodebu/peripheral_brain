Adding Time To A Date¶

In [7]:
# load the lubridate package
library(lubridate)
In [8]:
# create a date variable
date.ex <- dmy("1/1/2001")
In [9]:
date.ex
Out[9]:
[1] "2001-01-01 UTC"
In [10]:
# add 45 days to a date
date.ex.2 <- date.ex + days(45)
In [11]:
date.ex.2
Out[11]:
[1] "2001-02-15 UTC"
In [12]:
# add six weeks to a date
date.ex.3 <- date.ex + weeks(6)
In [13]:
date.ex.3
Out[13]:
[1] "2001-02-12 UTC"
